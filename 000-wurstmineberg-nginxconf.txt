server {
    root /usr/share/nginx/www/wurstmineberg.de;
    index index.html index.htm;
    
    server_name localhost;
    server_name 146.255.57.36;
    server_name wurstmineberg.de;
    server_name www.wurstmineberg.de;
    
    location / {
        try_files $uri $uri.html $uri.htm $uri.php $uri/ =404;
        ssi on;
    }
    
    location /people/ {
        ssi on;
        try_files /people/index.html =404;
    }
    
    location /static/ {
        proxy_pass http://assets.wurstmineberg.de/;
    }
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php5-fpm.sock;
        fastcgi_index index.php;
        include fastcgi_params;
    }
}
