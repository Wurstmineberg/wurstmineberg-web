server {
    root /usr/share/nginx/www/dev.wurstmineberg.de;
    index index.html index.htm;
    
    server_name dev.wurstmineberg.de;
    
    location / {
        try_files $uri $uri.html $uri.htm $uri.php $uri/ =404;
        ssi on;
    }
    
    location /doc/ {
        alias /usr/share/doc/;
        autoindex on;
        allow 127.0.0.1;
        allow ::1;
        deny all;
    }
    
    location /people/ {
        ssi on;
        try_files /people/index.html =404;
    }
    
    location /static/ {
        proxy_pass http://assets.wurstmineberg.de/
    }
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php5-fpm.sock;
        fastcgi_index index.php;
        include fastcgi_params;
    }
}
